package main

import "fmt"

type ListNode struct {
	Val  int
	Next *ListNode
}

//crete linklist
func CreateNode(node *ListNode, max int) {
	cur := node
	for i := 1; i < max; i++ {
		cur.Next = &ListNode{}
		cur.Next.Val = i
		cur = cur.Next
	}
}

//print linklist
func PrintNode(node *ListNode) {
	for cur := node; cur != nil; cur = cur.Next {
		fmt.Print(cur.Val, " ")
	}
	fmt.Println()
}

//leetcode 203. 移除链表元素
func removeElements(head *ListNode, val int) *ListNode {
	if head == nil {
		return head
	}
	head.Next = removeElements(head.Next, val)
	if head.Val == val {
		return head.Next
	}
	return head
}

func main() {
	var head = new(ListNode)
	CreateNode(head, 7)
	removeElements(head, 6)
	PrintNode(head)

}

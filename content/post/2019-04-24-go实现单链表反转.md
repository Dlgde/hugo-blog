单链表核心代码如下：

```go
func reverseList(node *ListNode) *ListNode {
	cur := node
	var pre *ListNode
	for cur != nil {
		cur.Next, pre, cur = pre, cur, cur.Next
	}
	return pre
}
```

理解：cur指向当前节点，pre是前节点初始为nil，反转过程：

```go
cur.next=pre
pre=cur
cur=cur.next
```

完整代码如下：
```go
package main

import "fmt"

type ListNode struct {
	data interface{}
	Next *ListNode
}

//反转单链表
func reverseList(head *ListNode) *ListNode {
	cur := head
	var pre *ListNode
	for cur != nil {
		cur.Next, pre, cur = pre, cur, cur.Next
	}
	return pre
}

func CreateNode(node *ListNode, max int) {
	cur := node
	for i := 1; i &lt; max; i++ {
		cur.Next = &ListNode{}
		cur.Next.data = i
		cur = cur.Next
	}
}

//打印链表的方法
func PrintNode(info string, node *ListNode) {
	fmt.Print(info)
	for cur := node; cur != nil; cur = cur.Next {
		fmt.Print(cur.data, " ")
	}
	fmt.Println()
}

func main() {
	var head = new(ListNode)
	CreateNode(head, 10)
	PrintNode("前：", head)
	yyy := reverseList(head)
	PrintNode("后：", yyy)

}
```
